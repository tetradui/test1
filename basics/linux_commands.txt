ls = показывает содержимое папки (файлы и др папки)
ls название_директории(py33) = показывает содержимое папки 
ls -a = показывает скрытые файлы и папки
ls -l = показывает доп информацию папок,файлов (владелец, дата, последнее открытие) 
ls -la = показывает скрытые файлы и директории с подробной информацией

cd = переход на дом.директорию
cd ~ = переход в дом.
cd test = переход в папку test, если такой папки нет, то ошибка 
cd .. = переход на папку выше 
cd ../.. = переход на две папки выше 
cd - = command+z

mkdir test = создает папку test в текущей папке
mkdir test/test2 = создает папку test2 внутри папки test, если папка существует 
mkdir test test2 = создает две папки test и test2 в текущей директории(папки) 

touch file.txt = создает файл file c разришение(форматом) ".txt"
touch file1.txt file2.txt = создает два файла "file1.txt" и "file2.txt"

rm test.txt = удаляет файлы 
rmdir test = удаляет пустую папку
rm -rf test = удаляет папку с содержимым (рекурсивное - полно удаление папки) 

mv test.txt test = перемещение файла (test.txt) в папку (test) 
mv test.txt test1.txt = переименование файла (test.txt на test1.txt)
 #mv работает также с папками 

pwd = показывет абсолютный путь (начиная с корневой папки) 
history = история введенных команд в терминале (одна сессия) 
history 20 = показывает 20 введенных команд ранее
history | grep test = выводвит все введенные команды содержащие слово test

nano test.txt = открывает файл test.txt в текстовом редакторе 'nano'
cat test.txt  = выводит содержимое файла в терминал


sudo(super user do)  = права суперюзера(linux)  
apt = системный установщик пакетов 
brew = установщик пакетов (macOs)
